 cd project_dir

Changed in working directory
 git status
Tracked file changes
 git diff
Add changed files
 git add file1 file2 file3
remove file
 git rm file
git rm dir/ -r (recursive under directory)
see files ready for commit
 git diff --cached
Commit changes
 git commit
git commit -m "My message"
git commit -a -m "My message"
(tracked files only, auto add)
Change last commit
 git commit --amend
Revert changes to file
 git checkout -- file

 git revert HEAD

 git log --pretty=short

 git log file

 git log dir/
